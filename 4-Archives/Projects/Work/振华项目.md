### 在线时长统计

```sql
CREATE TABLE IF NOT EXISTS zhenhua_zht.t_attendance (
    id bigserial,
    "time" timestamp with time zone,
    trackable_id character varying COLLATE pg_catalog."default",
    attendance integer,
    CONSTRAINT t_attendance_pkey PRIMARY KEY (id)
);

-- 创建 procedure
CREATE OR REPLACE PROCEDURE zhenhua_zht.online_time_count (job_id int, config jsonb)
LANGUAGE PLPGSQL
AS $
DECLARE
  ht REGCLASS;
  gap float;
  t_id uuid;
  count_time timestamptz;
  end_time timestamptz;
BEGIN
  SELECT jsonb_object_field_text (config, 'hypertable')::regclass INTO STRICT ht;
  SELECT jsonb_object_field_text (config, 'gap')::float INTO STRICT gap;
  SELECT jsonb_object_field_text (config, 'count_time')::timestamptz INTO STRICT count_time;
  SELECT jsonb_object_field_text (config, 'end_time')::timestamptz INTO STRICT end_time;

  IF gap IS NULL THEN
    RAISE EXCEPTION 'Config must have gap';
  END IF;

  IF count_time IS NULL THEN
      count_time = now();
  END IF;

  for t_id in
   select tt.id as t_id from zhenhua_zht.t_trackable as tt
  loop
    insert into zhenhua_zht.t_attendance(time, trackable_id, attendance)
    select count_time, t_id, count(*) as attendance from zhenhua_zht.t_position as tp
    where tp.device_time >= count_time - interval '1 day'
      and tp.device_time <= count_time
      and tp.trackable_id = t_id;
  end loop;
END
$;

-- 添加自动运行
SELECT add_job('zhenhua_zht.online_time_count','1d',
  initial_start=>'2022-03-22 24:00:00+08',
  config=>'{"gap":"1"}');
```
